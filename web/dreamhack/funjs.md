```javascript
// compute & print flag
(() => {
  const A = [0x20,0x5e,0x7b,0xd2,0x59,0xb1,0x34,0x72,0x1b,0x69,0x61,0x3c,0x11,0x35,0x65,0x80,0x09,0x9d,0x09,0x3d,0x22,0x7b,0x01,0x9d,0x59,0xaa,0x02,0x6a,0x53,0xa7,0x0b,0xcd,0x25,0xdf,0x01,0x9c];
  const B = [0x24,0x16,0x01,0xb1,0x0d,0x4d,0x01,0x13,0x1c,0x32,0x01,0x0c,0x20,0x02,0x01,0xe1,0x2d,0x6c,0x06,0x59,0x11,0x17,0x35,0xfe,0x0a,0x7a,0x32,0x0e,0x13,0x6f,0x05,0xae,0x0c,0x7a,0x61,0xe1];

  const ops = [
    (x,y) => x + y,
    (x,y) => x - y,
    (x,y) => x * y,
    (x,y) => x ^ y
  ];

  let out = '';
  for (let i = 0; i < A.length; i++) {
    const val = ops[i % ops.length](A[i], B[i]);
    out += String.fromCharCode(val);
  }

  console.log('FLAG:', out);
  return out;
})();
```
